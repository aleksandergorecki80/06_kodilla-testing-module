SELECT I.ID, I.SUMMARY, I.DESCRIPTION, IL.NAME AS LIST_NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID
ORDER BY I.ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.LASTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_COUNT
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY TASK_COUNT DESC;
